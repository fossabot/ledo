func (e SupportedRuntime) Command() string {
	switch e {
	case Docker:
		return "docker"
	case Podman:
		return "podman"
	default:
		return "docker"
	}
}

func (e SupportedRuntime) Compose() string {
	switch e {
	case Docker:
		return "docker-compose"
	case Podman:
		return "podman-compose"
	default:
		return "docker-compose"
	}
}
